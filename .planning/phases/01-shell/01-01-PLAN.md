**Step 1: Create the Todo directory and TodoItem.swift**

Create directory: `Peninsula/Todo/`

Create file `Peninsula/Todo/TodoItem.swift` with this exact content:

```swift
import Foundation

enum Priority: String, Codable, CaseIterable {
    case high
    case normal
    case low
}

struct TodoItem: Identifiable, Codable {
    var id: UUID
    var title: String
    var priority: Priority
    var isDone: Bool
    var createdAt: Date

    init(
        id: UUID = UUID(),
        title: String,
        priority: Priority = .normal,
        isDone: Bool = false,
        createdAt: Date = Date()
    ) {
        self.id = id
        self.title = title
        self.priority = priority
        self.isDone = isDone
        self.createdAt = createdAt
    }
}
```

**Step 3: Create and run the registration script**

Create file `add_todo_files.rb` in the project root with this content:

```ruby
require 'xcodeproj'

project_path = 'Peninsula.xcodeproj'
project = Xcodeproj::Project.open(project_path)

target = project.targets.find { |t| t.name == 'Peninsula' }
raise "Target 'Peninsula' not found" unless target

# Find the main Peninsula group
main_group = project.main_group['Peninsula']
raise "Main group 'Peninsula' not found" unless main_group

# Create or find the Todo group
todo_group = main_group['Todo'] || main_group.new_group('Todo', 'Peninsula/Todo')

# Add TodoItem.swift if not already present
unless todo_group.files.any? { |f| f.display_name == 'TodoItem.swift' }
  file_ref = todo_group.new_file('TodoItem.swift')
  target.source_build_phase.add_file_reference(file_ref)
  puts "Added TodoItem.swift to Peninsula target"
else
  puts "TodoItem.swift already registered"
end

project.save
puts "Project saved successfully"
```

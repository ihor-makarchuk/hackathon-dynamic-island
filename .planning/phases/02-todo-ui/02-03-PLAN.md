```swift
import SwiftUI

struct TodoCounterBadge: View {
    @ObservedObject var store = TodoStore.shared
    var notchHeight: CGFloat

    private var count: Int {
        store.incompleteCount(for: Date())
    }

    var body: some View {
        if count > 0 {
            Text("\(count)")
                .font(.system(size: notchHeight * 0.45, weight: .semibold, design: .rounded))
                .foregroundColor(.white.opacity(0.85))
                .frame(height: notchHeight)
        }
    }
}
```

The badge:
- Only renders when `count > 0` (disappears when no incomplete items today)
- Uses `notchHeight` parameter to proportionally size the font (45% of notch bar height)
- White text with slight transparency to match Peninsula's collapsed notch aesthetic
- Reads from `TodoStore.shared` so it reactively updates when items change

**1. Update `Peninsula/Notch/NotchViewModel.swift`:**

Add a `@ObservedObject var todoStore = TodoStore.shared` property alongside the existing `@ObservedObject var galleryModel`.

Add a computed property for the counter badge width:
```swift
var todoCounterWidth: CGFloat {
    if status == .opened { return 0 }
    let count = todoStore.incompleteCount(for: Date())
    if count == 0 { return 0 }
    // Width for the counter number + spacing
    return deviceNotchRect.height * 0.8 + deviceNotchRect.height / 8
}
```

Update `abstractSize` to include the counter width. The existing `abstractSize` already computes based on `notifModel.names.count`. Add the `todoCounterWidth` to the non-opened case. Locate the `abstractSize` computed property and update it:

```swift
var abstractSize: CGFloat {
    if status == .opened {
        return 0
    } else {
        var size: CGFloat = 0
        let count = notifModel.names.count
        if count > 0 {
            let spacing = deviceNotchRect.height / 8
            let itemHeight = deviceNotchRect.height
            size = spacing + (itemHeight + spacing) * CGFloat(count)
        }
        size += todoCounterWidth
        return size
    }
}
```

Import `Foundation` is already present. Add `import SwiftUI` if not already there (it should be via existing `@ObservedObject`).

Note: Because `TodoStore` is a class conforming to `ObservableObject`, using `@ObservedObject` on it in `NotchViewModel` (which is itself an `ObservableObject`) means the outer `NotchViewModel` won't automatically republish changes from `todoStore`. To fix this, add a Combine subscription in the `setupCancellables()` method or in init. Find the `setupCancellables()` method and add:

```swift
todoStore.objectWillChange
    .receive(on: DispatchQueue.main)
    .sink { [weak self] _ in self?.objectWillChange.send() }
    .store(in: &cancellables)
```

This ensures that when `TodoStore.items` changes (item added/deleted/toggled), `NotchViewModel` republishes so that all views observing it (including `NotchHoverView`) re-render and update the counter badge width and visibility.
